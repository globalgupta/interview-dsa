1. Two Sum — O(n²) 
let nums = [2,7,11,15]
let target = 9

for(let i=0; i<nums.length; i++){
    for(let j=i+1; j<nums.length; j++){
        if(nums[i] + nums[j] === target){
            console.log(i, j)
        }
    }
}
2. Valid Parentheses — Stack 
let str = "[{()}]"
let stack = []

for(let i=0; i<str.length; i++){

    let ch = str[i]

    if(ch == "[" || ch == "{" || ch == "("){
        stack.push(ch)
    }
    else{

        let last = stack.pop()

        if(
        (ch == "]" && last != "[") ||
        (ch == "}" && last != "{") ||
        (ch == ")" && last != "(")
        ){
            console.log("Invalid")
            return
        }

    }

}

if(stack.length == 0)
    console.log("Valid")
else
    console.log("Invalid")
3. Longest Substring Without Repeating Characters — O(n) 
let str = "abcabcbb"

let set = new Set()

let max = 0
let left = 0

for(let right = 0; right < str.length; right++){

    while(set.has(str[right])){

        set.delete(str[left])
        left++

    }

    set.add(str[right])

    max = Math.max(max, right - left + 1)

}

console.log(max)
