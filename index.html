1. Two Sum — O(n²) 

let nums = [2,7,11,15]
let target = 9

for(let i=0; i<nums.length; i++){
    for(let j=i+1; j<nums.length; j++){
        if(nums[i] + nums[j] === target){
            console.log(i, j)
        }
    }
}
2. Valid Parentheses — Stack 

let str = "[{()}]"
let stack = []

for(let i=0; i<str.length; i++){
    let ch = str[i]

    if(ch == "[" || ch == "{" || ch == "("){
        stack.push(ch)
    }
    else{
        let last = stack.pop()

        if(
        (ch == "]" && last != "[") ||
        (ch == "}" && last != "{") ||
        (ch == ")" && last != "(")
        ){
            console.log("Invalid")
            return
        }
    }
}

if(stack.length == 0)
    console.log("Valid")
else
    console.log("Invalid")
